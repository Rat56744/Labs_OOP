#      1, .. x_i - y_i > 1 -> +

#

    if ((black[i] - april[i]) > 1):
        r += 1
#        0  r
C = [math.comb(l, i) for i in range(r + 1)]
#   b,      1
b = (1 / 2) ** l * sum(C)
#    
if b <= alpha / 2:
    print("H1 ") #
