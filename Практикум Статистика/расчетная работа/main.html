def Ivan(N):
    I = [0]
    a = [1, 2, 3, 4, 5, 6]  # 1 -  , 0 -  
    x = 1
    p = [1/6] * 6  #    

    for i in range(N):
        x_n = np.random.choice(a, x, True, p)

        # ,     1    I[-1]
        if I[-1] >= 800:
            break

        match x_n[0]:  #  x_n[0]   
            case 1 | 2:
                I.append(I[-1] + 5)
            case 3 | 4:
                if I[-1] > 5:
                    I.append(I[-1] - 5)
            case 5 | 6:
                I.append(I[-1] + np.random.choice(a, x, True, p)[0] * 5)

    return I  #   I
