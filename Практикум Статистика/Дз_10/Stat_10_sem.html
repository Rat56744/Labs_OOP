M = [0, 0]  #     X  Y
K = [[1, 0], [0, 1]]  #   

r_pearson = []
r_spearman = []
p = []

#  
n = 3000

for i in range(n):
    #    X  Y
    x_y = np.random.multivariate_normal(M, K, 1000)
    X = x_y[:, 0]
    Y = x_y[:, 1]
    
    #  V  W
    V = X * Y  
    W = 0.5 * (X**2 - Y**2)  
    
    #        V  W
    pearson_corr = sts.pearsonr(V, W)
    spearman_corr = sts.spearmanr(V, W)
    
    #     
    r_pearson.append(pearson_corr[0])
    r_spearman.append(spearman_corr[0])
    p.append(pearson_corr[1])  # p-   

plt.hist(V,color='#9B2D30', bins= 'auto',edgecolor='#000000', density=True)
plt.hist(W,color='#1B2D30', bins= 'auto',edgecolor='#000000', density=True)

print("  :", sum(r_pearson) / n)
print("  :", sum(r_spearman) / n)
print(" p:", sum(p) / n)

#       alpha
alpha = sum(p) / n
