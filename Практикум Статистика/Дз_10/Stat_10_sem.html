# Определяем параметры многомерного нормального распределения
M = [0, 0]  # Вектор средних значений для X и Y
K = [[1, 0], [0, 1]]  # Ковариационная матрица 

r_pearson = []
r_spearman = []
p = []

# Количество экспериментов
n = 3000

for i in range(n):
    # Генерируем случайные величины X и Y
    x_y = np.random.multivariate_normal(M, K, 1000)
    X = x_y[:, 0]
    Y = x_y[:, 1]
    
    # Вычисляем V и W
    V = X * Y  
    W = 0.5 * (X**2 - Y**2)  
    
    # Расчет коэффициентов корреляции Пирсона и Спирмена для V и W
    pearson_corr = sts.pearsonr(V, W)
    spearman_corr = sts.spearmanr(V, W)
    
    # Добавляем элементы в конец списков
    r_pearson.append(pearson_corr[0])
    r_spearman.append(spearman_corr[0])
    p.append(pearson_corr[1])  # p-значение из теста Пирсона

plt.hist(V,color='#9B2D30', bins= 'auto',edgecolor='#000000', density=True)
plt.hist(W,color='#1B2D30', bins= 'auto',edgecolor='#000000', density=True)

print("Коэффициент корреляции Пирсона:", sum(r_pearson) / n)
print("Коэффициент корреляции Спирмена:", sum(r_spearman) / n)
print("Среднее p:", sum(p) / n)

# Проверка гипотезы при заданном уровне значимости alpha
alpha = sum(p) / n
print("При alpha <", alpha,", мы принимаем гипотезу Н0 ")
print(f"p значение для V  {sts.shapiro(V) }")
print(f"p значение для W  {sts.shapiro(V)}")
